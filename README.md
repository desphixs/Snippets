<h3>How to get current date or time in ajax jquery</h3>
<br>
var dt = new Date(); 
<br>
var time = dt.getDay() + " " + dt.getMonth() + "," + dt.getFullYear() + " , " + dt.getHours() + ":" + dt.getMinutes() + ":" + dt.getSeconds();; 
<br>
document.write(time);
<br>
